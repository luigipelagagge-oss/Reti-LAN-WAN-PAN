Esercizio 2: Sender & Receiver (Ping-Pong)

In questo esperimento facciamo dialogare due schede. Il Sender √® insistente e saluta ogni 2 secondi. Il Receiver √® paziente: conta i saluti e risponde "Grazie", ma alla decima volta chiude la conversazione con un "Ho capito".

1. Codice Sender (A)

// CODICE SENDER (Arduino A)

void setup() {
  Serial.begin(9600);
}

void loop() {
  // Invia il messaggio con accapo
  Serial.println("ciao"); 
  
  // Aspetta 2 secondi tra un invio e l'altro
  delay(2000);

2. Codice Receiver (B)

Conta i messaggi e reagisce.

// CODICE RECEIVER (Arduino B)

int contatore = 0; // Conta i messaggi

void setup() {
  Serial.begin(9600);
  pinMode(13, OUTPUT);
}

void loop() {
  // C'√® posta?
  if (Serial.available() > 0) {
    
    // Leggi e pulisci il messaggio
    String msg = Serial.readStringUntil('\n');
    msg.trim(); 

    // Se il messaggio √® "ciao"
    if (msg == "ciao") {
       contatore = contatore + 1; 

       // Feedback Visivo (Blink)
       digitalWrite(13, HIGH); 
       delay(100); 
       digitalWrite(13, LOW);

       // LOGICA DELLA RISPOSTA
       if (contatore < 10) {
           // Risposta standard (da 1 a 9)
           Serial.print("Grazie del messaggio ");
           Serial.println(contatore);
       } 
       else if (contatore == 10) {
           // Risposta finale (al 10)
           Serial.println("Ho capito");
       }
       // Se > 10 non fa pi√π nulla
    }
  }
}
